 data = app.DataTable;

            modelName = app.SelectAIModelDropDown.Value;

            switch modelName

                case "Decision Tree"
                    S = load("modelTree.mat");

                case "k-Nearest Neighbors"
                    S = load("modelKNN.mat");

                case "Support Vector Machine"
                    S = load("modelSVM.mat");

            end

            trainedModel = S.trainedModel;

            predictors = data(:,trainedModel.RequiredVariables);

            predictedStatus = trainedModel.predictFcn(predictors);

            data.PredictedStatus = predictedStatus;

            app.UITable2.Data = data;
            app.UITable2.ColumnName = data.Properties.VariableNames;

            app.DataTable = data;

            app.PredictionLabel.Text = "Prediction Completed";
